    # 🎉 elm-add-minus

    # 基本用法

    <template>
  <elm-add-minus :value="dynamic">
    <template v-slot:default="scope">
      <el-input v-model="scope.row.email"></el-input>
    </template>
  </elm-add-minus>
</template>

<script>
  export default {
    data() {
      return {
        dynamic: [
          {
            email: '',
          },
        ],
      }
    },
  }
</script>
